# App.js : 

    import Counter from './component/Counter'
    function App() {
    
      let emp = [
        {name : 'shad', age : 19},
        {name : 'John', age : 30},
        {name : 'mohamed', age : 25}
      ]
    
    
      return (
        <>
        {
          emp.map((obj) => {
    return(
      <Counter name={obj.name} age = {obj.age} />
    )
          })
        }
        </>
      )
    }
    
    export default App;

# Counter.js : 

    import React from 'react';
    
    function Counter(props) {
    
      return (
        <>
        <h1>Name : {props.name} ,  Age : {props.age}</h1>
        </>
      );
    }
    
    export default Counter;
